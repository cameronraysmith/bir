\begin{block}{expression of $L$ in terms of a coequalizer}
The functor $L_A$ can also be constructed as a coequalizer (colimits can generally be expressed in terms of coequalizers)
\begin{displaymath}
\xymatrix{
\coprod\limits_{\substack{u : L'_0 \rightarrow L_0 \\ p \in P(L_0)}}
A(L'_0)
\ar@<1ex>[r]^-{\theta} \ar@<-1ex>[r]_-{\tau}
&
\coprod\limits_{\substack{L_0 \\ p \in P(L_0)}}
A(L_0)
\ar[r]
&
P \otimes_{\mathcal{L}_0} A
}
\end{displaymath}
where $P \otimes_{\mathcal{L}_0} A = L_A(P)$.
\end{block}
\end{frame}
